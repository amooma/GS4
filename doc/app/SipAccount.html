272:     logger.debug "Trying to find Cantina SIP account for \"#{sip_user}@#{sip_server}\" ..."
273:     begin
274:       cantina_resource = provisioning_server_base_url( the_sip_phone_id )

287:         if cantina_sip_accounts

290:             logger.debug cantina_sip_account.inspect
291:             logger.debug "-------------}"
292:             if (cantina_sip_account.registrar .to_s == sip_server .to_s)              && (cantina_sip_account.user      .to_s == sip_user   .to_s)
293:               logger.debug "Found CantinaSipAccount ID #{cantina_sip_account.id}."
294:               return cantina_sip_account
295:               break
296:             end
297:           }
298:         end
299:         logger.debug "CantinaSipAccount for \"#{sip_user}@#{sip_server}\" not found."
300:         return nil
301:       end
302:     rescue Errno::ECONNREFUSED, Errno::EADDRNOTAVAIL, Errno::EHOSTDOWN, ActiveResource::TimeoutError => e

provisioning_server_base_url( the_sip_phone_id=nil ) click to toggle source

Returns the appropriate “site” parameter URL to use for the CantinaSipAccount ActiveResource. Returns nil if the SipPhone corresponding to this SipAccount does not have provisioning or if the SipAccount does not even have a SipPhone (and thus does not have a provisioning server). the_sip_phone_id can be nil to use the provisioning_server_id of the account’s current sip_phone, or you may want to pass in sip_phone_id_was. 

     # File app/models/sip_account.rb, line 226
226:   def provisioning_server_base_url( the_sip_phone_id=nil )
227:     scheme = 'http'
228:     host   = '0.0.0.0'  # this will not work on purpose
229:     port   = 0          # this will not work on purpose
230:     path   = '/'
231:     
232:     the_sip_phone_id = self.sip_phone_id if the_sip_phone_id == nil
233:     begin
234:       the_sip_phone = the_sip_phone_id ? SipPhone.find( the_sip_phone_id ) : nil
235:     rescue ActiveResource::ResourceNotFound => e
236:       the_sip_phone = nil
237:     end
238:     
239:     if the_sip_phone
240:       if the_sip_phone.provisioning_server_id.blank?
241:         logger.debug "SipPhone #{the_sip_phone_id.inspect} has no provisioning server. Alright."
242:         return nil  # phone without provisioning
243:       else
244:         the_prov_server = the_sip_phone.provisioning_server  # We're using a variable name different from the provisioning_server method.
245:         if the_prov_server
246:           scheme = 'http'  # TODO - https as soon as it is implemented.
247:           host   = the_prov_server.name if ! the_prov_server.name.blank?
248:           port   = the_prov_server.port if ! the_prov_server.port.blank?
249:           path   = '/'
250:         end
251:       end
252:     else
253:       logger.debug "SipAccount ID #{self.id} is not associated to a SipPhone, i.e. does not have a provisioning server."
254:       return nil  # SIP account without phone and thus without provisioning server
255:     end
256:     ret = '%s://%s%s%s' % [
257:       scheme,
258:       host,
259:       port.blank? ? '' : ":#{port.to_s}",
260:       path,
261:     ]
262:     logger.debug "SipPhone #{the_sip_phone_id.inspect} has prov. server #{ret.inspect}."
263:     return ret
264:   end

     # File app/models/sip_account.rb, line 172
172:         def cantina_find_sip_account_by_server_and_user( sip_server, sip_user )
173:                 logger.debug "Trying to find Cantina SIP account for \"#{sip_user}@#{sip_server}\" ..."
174:                 begin
175:                         cantina_resource = determine_prov_server_resource()
176:                         if ! cantina_resource
177:                                 # SipPhone has no provisioning server.
178:                                 return nil
179:                         else

183:                                 if cantina_sip_accounts
184:                                         cantina_sip_accounts.each { |cantina_sip_account|
185:                                                 if cantina_sip_account.sip_proxy .to_s == sip_server .to_s                                                  && cantina_sip_account.user      .to_s == sip_user   .to_s
186:                                                         logger.debug "Found CantinaSipAccount ID #{cantina_sip_account.id}."
187:                                                         return cantina_sip_account

189:                                                 end
190:                                         }
191:                                 end

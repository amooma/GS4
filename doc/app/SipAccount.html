Find a SIP account by SIP server and SIP user on the Cantina provisioning server. Returns the CantinaSipAccount if found or nil if not found or false on error. 

     # File app/models/sip_account.rb, line 175
175:   def cantina_find_sip_account_by_server_and_user( sip_server, sip_user )
176:     logger.debug "Trying to find Cantina SIP account for \"#{sip_user}@#{sip_server}\" ..."
177:     begin
178:       cantina_resource = determine_prov_server_resource()
179:       if ! cantina_resource
180:         # SipPhone has no provisioning server.
181:         return nil
182:       else
183:         CantinaSipAccount.set_resource( cantina_resource )
184:         cantina_sip_accounts = CantinaSipAccount.all()
185:         # GET "/sip_accounts.xml" - #OPTIMIZE - The Cantina API does not let us do more advanced queries so we have to get all SIP accounts.
186:         if cantina_sip_accounts
187:           cantina_sip_accounts.each { |cantina_sip_account|
188:             if cantina_sip_account.registrar .to_s == sip_server .to_s                && cantina_sip_account.auth_user      .to_s == auth_name   .to_s
189:               logger.debug "Found CantinaSipAccount ID #{cantina_sip_account.id}."
190:               return cantina_sip_account
191:               break
192:             end
193:           }
194:         end
195:         logger.debug "CantinaSipAccount not found."
196:         return nil
197:       end
198:     rescue Errno::ECONNREFUSED => e
199:       return false
200:     end
201:   end

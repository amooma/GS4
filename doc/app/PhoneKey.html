34:       case self.phone_key_function_definition.type_of_class
35:         when 'string'
36:           if ! self.value.class.ancestors.include?( String )

38:           end
39:         when 'integer'

42:           end
43:         when 'boolean'
44:           if ! self.value.match( /^(?:true|false)$/ )
45:             errors.add( :value, "(\"#{self.value}\") for \"#{self.phone_key_function_definition.name}\" softkey has to be boolean (\"true\" or \"false\")." )
46:           end
47:         when 'url'
48:           require 'uri'
49:             begin
50:               uri = URI.parse( self.value )
51:               # Require an absolute URI (i.e. with a scheme):
52:               if ! uri.absolute?

54:               end
55:               rescue URI::InvalidURIError
56:                 errors.add( :value, "(\"#{self.value}\") for \"#{self.phone_key_function_definition.name}\" softkey has to be an URL." )
57:             end
58:         when 'uri'

64:               if ! uri.fragment.nil?
65:                 errors.add( :value, "(\"#{self.value}\") for \"#{self.phone_key_function_definition.name}\" softkey has to be an URI (URI's don't have a fragment)." )
66:               end
67:               rescue URI::InvalidURIError
68:                 errors.add( :value, "(\"#{self.value}\") for \"#{self.phone_key_function_definition.name}\" softkey has to be an URI." )
69:             end

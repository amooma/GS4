sip_displayname_encode( str ) click to toggle source

Encode special characters in a SIP display-name. See the “display-name”/“qdtext”/“quoted-pair” rules in tools.ietf.org/html/rfc3261 . 

     # File app/controllers/freeswitch_call_processing_controller.rb, line 299
299:         def sip_displayname_encode( str )
300:                 #orig_str = str
301:                 str = str.to_s.force_encoding( Encoding::ASCII_8BIT )
302:                 
303:                 # Sanitize control characters. They could be encoded but cause
304:                 # problems for many implementations:
305:                 str = str.gsub( /[\x00-\x1F]+/, ' ' )
306:                 
307:                 # FreeSwitch doesn't handle "<", ">" even in its encoded forms:
308:                 str = str.gsub( /[<>]+/, ' ' )
309:                 
310:                 # without "/", ":", "<", ">" (which are special for FreeSwitch)
311:                 safe = /(?:
312:                           [ !\#\$\%&'()*+,\-.0-9;=?@A-Z\[\]^_`a-z{|}~]
313:                         | [\xC0-\xDF][\x80-\xBF]{1}
314:                         | [\xE0-\xEF][\x80-\xBF]{2}
315:                         | [\xF0-\xF7][\x80-\xBF]{3}
316:                         | [\xF8-\xFB][\x80-\xBF]{4}
317:                         | [\xFC-\xFD][\x80-\xBF]{5}
318:                 )+/n
319:                 ret = []
320:                 str = str.to_s << 'a'  # add a safe char at the end so we get a match
321:                 pos = 0
322:                 while ((safe_part = safe.match( str, pos )) != nil)
323:                         # add the unsafe part before the current match:
324:                         #ret << "-" + "{" + str[ pos, safe_part.begin(0)-pos ] + "}"
325:                         ret << percent_encode_re( str[ pos, safe_part.begin(0)-pos ], /./ )
326:                         
327:                         # add the safe part:
328:                         ret << safe_part.to_s
329:                         
330:                         pos = safe_part.end(0)
331:                 end
332:                 ret = ret.join('')
333:                 ret = ret[ 0, ret.bytesize-1 ].force_encoding( Encoding::UTF_8 )
334:                 #logger.info( "##############     #{orig_str}  =>  #{ret}    " )
335:                 return ret
336:         end

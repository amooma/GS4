405:         def sip_displayname_encode( str )
406:                 #orig_str = str
407:                 str = str.to_s.force_encoding( Encoding::ASCII_8BIT )
408:                 
409:                 # Sanitize control characters. They could be encoded but cause
410:                 # problems for many implementations:
411:                 str = str.gsub( /[\x00-\x1F]+/, ' ' )

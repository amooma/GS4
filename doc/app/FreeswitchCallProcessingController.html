sip_displayname_encode( str ) click to toggle source

Encode special characters in a SIP display-name. See the “display-name”/“qdtext”/“quoted-pair” rules in tools.ietf.org/html/rfc3261 . 

     # File app/controllers/freeswitch_call_processing_controller.rb, line 378
378:         def sip_displayname_encode( str )
379:                 #orig_str = str
380:                 str = str.to_s.force_encoding( Encoding::ASCII_8BIT )
381:                 
382:                 # Sanitize control characters. They could be encoded but cause
383:                 # problems for many implementations:
384:                 str = str.gsub( /[\x00-\x1F]+/, ' ' )
385:                 
386:                 # FreeSwitch doesn't handle "<", ">" even in its encoded forms:
387:                 str = str.gsub( /[<>]+/, ' ' )
388:                 
389:                 # without "/", ":", "<", ">" (which are special for FreeSwitch)
390:                 safe = /(?:
391:                           [ !\#\$\%&'()*+,\-.0-9;=?@A-Z\[\]^_`a-z{|}~]
392:                         | [\xC0-\xDF][\x80-\xBF]{1}
393:                         | [\xE0-\xEF][\x80-\xBF]{2}
394:                         | [\xF0-\xF7][\x80-\xBF]{3}
395:                         | [\xF8-\xFB][\x80-\xBF]{4}
396:                         | [\xFC-\xFD][\x80-\xBF]{5}
397:                 )+/n
398:                 ret = []
399:                 str = str.to_s << 'a'  # add a safe char at the end so we get a match
400:                 pos = 0
401:                 while ((safe_part = safe.match( str, pos )) != nil)
402:                         # add the unsafe part before the current match:
403:                         #ret << "-" + "{" + str[ pos, safe_part.begin(0)-pos ] + "}"
404:                         ret << percent_encode_re( str[ pos, safe_part.begin(0)-pos ], /./ )
405:                         
406:                         # add the safe part:
407:                         ret << safe_part.to_s
408:                         
409:                         pos = safe_part.end(0)
410:                 end
411:                 ret = ret.join('')
412:                 ret = ret[ 0, ret.bytesize-1 ].force_encoding( Encoding::UTF_8 )
413:                 #logger.debug( "sip_displayname_encode( #{orig_str.inspect} )  =>  #{ret.inspect}" )
